Pythoon3.x对豆瓣TOP250电影爬虫

Posted on 2017-03-17 | In Python , 未分类

一直想要学习爬虫,看了两天Python书籍,把基础数据类型和基础语法了解了一下.

在网上找了学习资料和BeautifulSoup4中文说明,链接在下方给出.

主要使用三个模块requests/BeautifulSoup4/pymysql

要抓取数据的网页为 https://movie.douban.com/top250

代码和解释都挺清晰了

<span class="hljs-comment">#! encoding = utf-8</span>
<span class="hljs-keyword">import</span> requests
<span class="hljs-keyword">from</span> bs4 <span class="hljs-keyword">import</span> BeautifulSoup
<span class="hljs-keyword">import</span> pymysql
<span class="hljs-keyword">import</span> traceback

base_url = <span class="hljs-string">"https://movie.douban.com/top250"</span>
conn = pymysql.connect(<span class="hljs-string">"localhost"</span>, <span class="hljs-string">"root"</span>, <span class="hljs-string">"123456"</span>, <span class="hljs-string">"test"</span>, use_unicode=<span class="hljs-keyword">True</span>, charset=<span class="hljs-string">"utf8"</span>)
cursor = conn.cursor()

<span class="hljs-function"><span class="hljs-keyword">def</span> <span class="hljs-title">save_to_database</span><span class="hljs-params">(movie_id, movie_title, movie_image, movie_des, movie_quote)</span>:</span>
    sql = <span class="hljs-string">'insert into douban(movie_id,movie_title,movie_image,movie_description,movie_quote) VALUES ("%s","%s","%s","%s","%s")'</span> % \
          (movie_id, movie_title, movie_image, movie_des, movie_quote)

    <span class="hljs-keyword">try</span>:
        cursor.execute(sql)
        conn.commit()
        print(<span class="hljs-string">"success"</span>)
    <span class="hljs-keyword">except</span>:
        conn.rollback()
        traceback.print_exc()

<span class="hljs-function"><span class="hljs-keyword">def</span> <span class="hljs-title">get_list</span><span class="hljs-params">(url)</span>:</span>
    <span class="hljs-keyword">global</span> base_url
    req = requests.get(url).text
    soup = BeautifulSoup(req, <span class="hljs-string">"html.parser"</span>)
    movie_list = soup.find_all(<span class="hljs-string">"ol"</span>, <span class="hljs-string">"grid_view"</span>)
    <span class="hljs-keyword">for</span> item <span class="hljs-keyword">in</span> movie_list:
        <span class="hljs-keyword">for</span> detail <span class="hljs-keyword">in</span> item.find_all(<span class="hljs-string">'li'</span>):
            movie_id = str(detail.find(<span class="hljs-string">'em'</span>).text)
            image_url = str(detail.img.attrs[<span class="hljs-string">'src'</span>])
            title = str(detail.find(<span class="hljs-string">'span'</span>, <span class="hljs-string">'title'</span>).text)
            description = str(detail.find(<span class="hljs-string">'div'</span>, <span class="hljs-string">'bd'</span>).p.text)
            <span class="hljs-keyword">try</span>:
                quote = str(detail.find(<span class="hljs-string">'p'</span>, <span class="hljs-string">'quote'</span>).text)
            <span class="hljs-keyword">except</span>:
                quote = <span class="hljs-string">'暂无引用'</span>
            <span class="hljs-comment"># 测试打印的数据</span>
            <span class="hljs-comment"># print(movie_id, title, image_url, description, quote)</span>
            <span class="hljs-comment"># 进行数据库插入</span>
            save_to_database(movie_id, title, image_url, description, quote)

    <span class="hljs-comment"># 下一页地址</span>
    next_page = soup.find(<span class="hljs-string">'span'</span>, attrs={<span class="hljs-string">'class'</span>: <span class="hljs-string">'next'</span>}).find(<span class="hljs-string">'a'</span>)
    <span class="hljs-keyword">if</span> next_page:
        new_url = base_url + next_page[<span class="hljs-string">'href'</span>]
        <span class="hljs-comment"># 使用递归,重复调用此函数,直到没有新的数据页</span>
        get_list(new_url)
        <span class="hljs-keyword">return</span> base_url
    <span class="hljs-keyword">else</span>:
        base_url = <span class="hljs-keyword">None</span>
    <span class="hljs-keyword">return</span> <span class="hljs-keyword">None</span>

get_list(base_url)
